Swift Cheat Sheet

A clean and clear cheat sheet and guide for Apple's Swift language

Variables

var myString = "hello"
var myExplicitString: String = "hello" // explicit type
var x = 1, y = 2, z = 3 // declare multiple integers
myExplicitString = "goodbye" // set to another string value

Constants

let myString = "hello"
myString = "how are you" // compile-time error

Printing

let name = "swift"
println("Hello")
println("My name is \(name)")
print("See you "); 
print("later")
/*  Hello
    My name is swift
    See you later */
